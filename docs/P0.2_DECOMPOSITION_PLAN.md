class RendererCore {

    bool initialize();

    vk::Device getDevice() const { return device_; }
    vk::Queue getGraphicsQueue() const { return graphicsQueue_; }
    vk::Queue getPresentQueue() const { return presentQueue_; }
    VmaAllocator getAllocator() const { return allocator_; }

    bool createInstance();
    bool pickPhysicalDevice();
    bool createLogicalDevice();
    bool createAllocator();
    
    vk::Instance instance_;
    vk::PhysicalDevice physicalDevice_;
    vk::Device device_;
    vk::Queue graphicsQueue_;
    vk::Queue presentQueue_;

};

// ========== 2. SwapchainManager (~180 строк) ==========
class SwapchainManager {
public:
    struct Config {
        vk::Device device;
        vk::PhysicalDevice physicalDevice;
        vk::SurfaceKHR surface;
        uint32_t width;
        uint32_t height;
    };
    
    bool initialize(const Config& config);
    void shutdown();
    void recreate(uint32_t width, uint32_t height);
    
    vk::SwapchainKHR getSwapchain() const { return swapchain_; }
    const std::vector<vk::Image>& getImages() const { return swapchainImages_; }
    const std::vector<vk::ImageView>& getImageViews() const { return swapchainImageViews_; }
    vk::Format getFormat() const { return swapchainFormat_; }
    vk::Extent2D getExtent() const { return swapchainExtent_; }
    
private:
    bool createSwapchain();
    bool createImageViews();
    void destroySwapchain();
    
    vk::Device device_;
    vk::PhysicalDevice physicalDevice_;
    vk::SurfaceKHR surface_;
    vk::SwapchainKHR swapchain_;
    std::vector<vk::Image> swapchainImages_;
    std::vector<vk::ImageView> swapchainImageViews_;
    vk::Format swapchainFormat_;
    vk::Extent2D swapchainExtent_;
    uint32_t width_;
    uint32_t height_;
};

// ========== 3. FrameManager (~180 строк) ==========
class FrameManager {
public:
    struct FrameResult {
        bool success;
        uint32_t imageIndex;
        bool needsRecreate;
    };
    
    bool initialize(vk::Device device, uint32_t maxFramesInFlight);
    void shutdown();
    
    FrameResult beginFrame(vk::SwapchainKHR swapchain);
    bool endFrame(vk::SwapchainKHR swapchain, 
                  vk::Queue graphicsQueue,
                  vk::Queue presentQueue, 
                  vk::CommandBuffer commandBuffer);
    
    size_t getCurrentFrame() const { return currentFrame_; }
    uint32_t getCurrentImageIndex() const { return currentImageIndex_; }
    bool isDeviceLost() const { return deviceLost_; }
    
    vk::Semaphore getImageAvailableSemaphore() const;
    vk::Semaphore getRenderFinishedSemaphore() const;
    vk::Fence getInFlightFence() const;
    
private:
    vk::Device device_;
    std::vector<vk::Semaphore> imageAvailableSemaphores_;
    std::vector<vk::Semaphore> renderFinishedSemaphores_;
    std::vector<vk::Fence> inFlightFences_;
    size_t currentFrame_ = 0;
    uint32_t currentImageIndex_ = 0;
    uint32_t maxFramesInFlight_ = 2;
    bool deviceLost_ = false;
};

class HybridFreGSRenderer final : public IRenderer {
public:
    // Constructor injection (DIP)
    HybridFreGSRenderer(
        std::shared_ptr<RendererCore> core,
        std::shared_ptr<SwapchainManager> swapchain,
        std::shared_ptr<FrameManager> frameManager,
        std::shared_ptr<PipelineManager> pipelineManager,
        std::shared_ptr<RendererDebugger> debugger,
        std::shared_ptr<RendererStatistics> statistics
    );
    
    ~HybridFreGSRenderer() override;
    
    // IRenderer
    bool initialize() override;
    bool attachWindow(void* x11Display, void* x11Window, uint32_t width, uint32_t height);
    void beginFrame() override;
    void renderFrame(const FrameData& frameData) override;
    void endFrame() override;
    void shutdown() override;
    
    // Data upload
    void uploadGaussians(const std::vector<GaussianSplat>& gaussians);
    void uploadTriangles(const std::vector<Triangle>& triangles);
    void uploadMesh(const std::shared_ptr<Mesh3D>& mesh);
    
    // Camera
    void setCamera(Camera3D* camera);
    
    // Render mode
    void setRenderMode(RenderMode mode);
    
    // Debug API (delegates to RendererDebugger)
    void setDebugMode(int mode) override;
    int getDebugMode() const override;
    void enableWireframe(bool enable) override;
    // ... остальные debug методы
    
    // Statistics (delegates to RendererStatistics)
    RenderingStats getStats() const override;
    DetailedRenderingStats getDetailedStats() const override;
    GPUInfo getGPUInfo() const override;
    
private:
    // Dependencies (DIP)
    std::shared_ptr<RendererCore> core_;
    std::shared_ptr<SwapchainManager> swapchain_;
    std::shared_ptr<FrameManager> frameManager_;
    std::shared_ptr<PipelineManager> pipelineManager_;
    std::shared_ptr<RendererDebugger> debugger_;
    std::shared_ptr<RendererStatistics> statistics_;
    
    // Render passes
    std::unique_ptr<TriangleSplattingPass> triangleSplattingPass_;
    std::unique_ptr<FreGSPass> fregsPass_;
    
    // State
    bool initialized_ = false;
    RenderMode renderMode_ = RenderMode::TriangleSplatting;
    Camera3D* camera_ = nullptr;
    
    // Helper methods
    void recordCommandBuffer(vk::CommandBuffer cmd, uint32_t imageIndex);
    glm::mat4 getCorrectedViewProjMatrix() const;
    std::vector<Triangle> convertMeshToTriangles(const Mesh3D& mesh);
};
